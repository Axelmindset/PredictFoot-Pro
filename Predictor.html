// ========== VARIABLES GLOBALES ========== var profiles = { a: { weather: 'normal', context: 'normal', forme: 'neutre', dynamique: 'super_faibles', reaction: 'attaque_subit' }, b: { weather: 'normal', context: 'normal', forme: 'neutre', dynamique: 'super_faibles', reaction: 'attaque_subit' } }; var categories = ['weather', 'context', 'forme', 'dynamique', 'reaction']; var pointsMap = { 'pluie': 4, 'soleil': 3, 'normal': 2, 'neutre': 1, 'decisif': 4, 'survie': 3, 'avant_treve': 1, 'apres_treve': -2, 'bonne_serie': 3, 'neutre': 1, 'irregulier': 0, 'crise': -1, 'super_forts': 3, 'top5': 1, 'super_faibles': 0, 'contre_perf': -1, 'reagit_vite': 4, 'subit_reagit': 2, 'attaque_subit': 0, 'subit_apres': -1, 'jamais': -2 }; // ========== INITIALISATION ========== function initProfileSelectors() { categories.forEach(cat => { setupTeamListeners('a', cat); setupTeamListeners('b', cat); }); } function setupTeamListeners(team, cat) { const container = document.getElementById(cat + team.toUpperCase() + '_options'); if (!container) return; const options = container.querySelectorAll('.profile-option'); options.forEach(opt => { opt.addEventListener('click', function() { options.forEach(o => o.classList.remove('selected')); this.classList.add('selected'); profiles[team][cat] = this.getAttribute('data-value'); }); }); } // ========== LOGIQUE DE PRÉDICTION (CORRIGÉE) ========== function getScoreFromPoints(pts) { // Logique de puissance basée sur les points réels if (pts >= 14) return Math.floor(Math.random() * 2) + 3; // 3 ou 4 buts if (pts >= 9) return Math.floor(Math.random() * 2) + 2; // 2 ou 3 buts if (pts >= 5) return Math.floor(Math.random() * 2) + 1; // 1 ou 2 buts if (pts >= 1) return Math.random() > 0.5 ? 1 : 0; // 0 ou 1 but return 0; } function generatePredictions() { var ptsA = 0, ptsB = 0; categories.forEach(c => { ptsA += pointsMap[profiles.a[c]] || 0; }); categories.forEach(c => { ptsB += pointsMap[profiles.b[c]] || 0; }); var nameA = document.getElementById('teamA').value || 'Équipe A'; var nameB = document.getElementById('teamB').value || 'Équipe B'; var gap = ptsA - ptsB; // Score 1 (Basé sur la forme actuelle) var s1A = getScoreFromPoints(ptsA); var s1B = getScoreFromPoints(ptsB); // Score 2 (Scénario alternatif basé sur l'écart de niveau) var s2A = s1A, s2B = s1B; if (gap > 5) { s2A += 1; s2B = Math.max(0, s2B - 1); } else if (gap < -5) { s2B += 1; s2A = Math.max(0, s2A - 1); } else { s2A = s1A + 1; s2B = s1B + 1; } // Scénario plus offensif // Probabilités dynamiques var totalButsS1 = s1A + s1B; var btts = (s1A > 0 && s1B > 0) ? 'OUI' : 'NON'; var bttsPercent = (btts === 'OUI') ? Math.min(88, 50 + (ptsA + ptsB) * 2) : 35; var over25 = (totalButsS1 >= 3) ? 'OUI' : 'NON'; var over25Percent = Math.min(92, (totalButsS1 * 20) + (ptsA + ptsB)); var under35 = (totalButsS1 <= 3) ? 'OUI' : 'NON'; var under35Percent = (totalButsS1 <= 2) ? 82 : (totalButsS1 === 3 ? 55 : 20); // 1N2 et Double Chance var winner = null, dc = null; if (Math.abs(gap) >= 8) winner = gap > 0 ? nameA : nameB; else dc = gap >= 2 ? nameA + "/N" : (gap <= -2 ? nameB + "/N" : "1/2 (Pas de Nul)"); // Confiance (Plafonnée à 95%) var conf = Math.min(95, 50 + (Math.abs(gap) * 4)); return { nameA, nameB, ptsA, ptsB, gap, s1A, s1B, s2A, s2B, btts, bttsPercent, over25, over25Percent, under35, under35Percent, winner, dc, conf, safeBet: (over25Percent > 75) ? 'Plus de 2.5 Buts' : (under35Percent > 75 ? 'Moins de 3.5 Buts' : 'Double Chance'), safePercent: Math.max(over25Percent, under35Percent, 65) }; } // ========== APPEL DES FONCTIONS ========== window.onload = function() { initProfileSelectors(); initTabs(); initTheme(); document.getElementById('analyserBtn').addEventListener('click', runAnalysis); }; function runAnalysis() { var data = generatePredictions(); displayResults(data); showToast('✅ Analyse mise à jour !'); } // (Garder les fonctions displayResults, showToast et copyWhatsApp de ton code initial)
